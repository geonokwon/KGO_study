String star = "*";
String space = " ";
int roop = scanner.nextInt();
for(int i = 1; i < roop; i++) {
	for(int j = 0; j < roop - i; j++) {
		System.out.print(space);
	}
	for(int k = 0; k < i * 2 - 1; k++) {
		System.out.print(star);
	}
	System.out.println();
}
for(int i = roop; i > 0; i--) {
	for(int k = roop - i; k > 0; k--) {
		System.out.print(space);
	}
	for(int j = i * 2 - 1; j > 0; j--) {
		System.out.print(star);
	}
	System.out.println();
}
